```python
class MultilingualTensorSystem:

    Complete multilingual tensor processing system.
    """
    def __init__(self, supported_languages):
        self.languages = supported_languages
        self.semantic_encoders = {lang: SemanticEncoder(lang) 
                                   for lang in supported_languages}
        self.universal_core = UniversalTensorCore()
        self.translators = {(l1, l2): Translator(l1, l2) 
                            for l1 in supported_languages 
                            for l2 in supported_languages}
    
    def process(self, tensor, source_lang, target_lang=None):
        # Encode with source language semantics
        encoded = self.semantic_encoders[source_lang].encode(tensor)
        
        # Transform to universal representation
        universal = self.universal_core.to_universal(encoded, source_lang)
        
        if target_lang and target_lang != source_lang:
            # Translate to target language semantics
            target_semantic = self.translators[(source_lang, target_lang)]
            return target_semantic.decode(universal)
        
        return universal
```

```python
        universal_part = self.universal_op.extract(tensor)

        local_part = self.local_encoders[language_context].decode(tensor)
        
        # Apply universal operation
        universal_result = self.universal_op(universal_part)
        
        # Reintegrate with localized semantics
        return self.local_encoders[language_context].encode(
            universal_result, local_part
        )
```
